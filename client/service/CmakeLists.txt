project(client-service)

message(STATUS "${PROJECT_NAME} is building")

aux_source_directory(${CMAKE_CURRENT_SOURCE_DIR}/src SRC)

if(SRC)
    add_library(${PROJECT_NAME} SHARED ${SRC})
    target_include_directories(${PROJECT_NAME}
        PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include
        PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/Impl

        PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}../business/core/include
        PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}../business/network/include
    )

    target_link_directories(${PROJECT_NAME}
    PRIVATE
        client-mapper
        client-core
        client-network
    )
else()
    message(WARNING "No source files found for ${PROJECT_NAME}")
endif()
